    4. React array map method 

    import React from 'react';

const Gallery = () => {
const data=[
    {
        id:1,
        img:"https://picsum.photos/210/300",
    },
    {
        id:2,
        img:"https://picsum.photos/210/300",
    },
];


const arraydata = data.map((val, i)=>{
    return(
        <img src={data[i].img}/>
    );
    
});

return (
    <>
        <h1>Gallery</h1>
        <div className="gallery">
            {arraydata}
        </div>
    </>
)
}

export default Gallery;
